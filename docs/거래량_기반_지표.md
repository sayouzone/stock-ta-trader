# 거래량 기반 지표 (OBV, VWAP, MFI) 시장 분석 방법

## 거래량 분석의 핵심 원칙

```
"가격은 거짓말을 하지만, 거래량은 거짓말을 하지 않는다"

가격 상승 + 거래량 증가  →  강한 상승 추세 확인  ✅
가격 상승 + 거래량 감소  →  추세 약화 / 분산 경고 ⚠️
가격 하락 + 거래량 증가  →  강한 하락 추세 확인  ✅
가격 하락 + 거래량 감소  →  하락 모멘텀 약화     ⚠️
```

## 지표 1: OBV (On-Balance Volume)

### 개념 및 계산

```python
def compute_obv(df: pd.DataFrame) -> pd.DataFrame:
    """
    종가 방향에 따라 거래량을 누적 합산
    상승봉: +Volume  /  하락봉: -Volume  /  보합: ±0

    OBV 방향 = 수요(매수)/공급(매도) 압력의 누적 흐름
    """
    df   = df.copy()
    obv  = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])

    df['obv']        = obv
    df['obv_ema20']  = pd.Series(obv).ewm(span=20, adjust=False).mean().values
    df['obv_slope']  = pd.Series(obv).diff(5)   # 5봉 변화량
    return df
```

### OBV 신호 분류

```python
def analyze_obv(df: pd.DataFrame) -> dict:
    obv      = df['obv']
    obv_ema  = df['obv_ema20']
    close    = df['close']

    cur_obv  = float(obv.iloc[-1])
    cur_ema  = float(obv_ema.iloc[-1])
    cur_close = float(close.iloc[-1])

    # ① OBV vs EMA 위치
    obv_vs_ema = "bullish" if cur_obv > cur_ema else "bearish"

    # ② OBV 추세 방향 (20봉 선형 기울기)
    x      = list(range(20))
    y      = obv.iloc[-20:].values
    slope  = float(np.polyfit(x, y, 1)[0])
    obv_trend = (
        "strong_up"   if slope > 0 and slope > abs(slope) * 0.5 else
        "mild_up"     if slope > 0 else
        "strong_down" if slope < 0 and abs(slope) > abs(slope) * 0.5 else
        "mild_down"
    )

    # ③ 가격-OBV 다이버전스
    price_20_high = close.iloc[-20:].max()
    price_20_low  = close.iloc[-20:].min()
    obv_20_high   = obv.iloc[-20:].max()
    obv_20_low    = obv.iloc[-20:].min()

    divergence = "none"
    # 약세 다이버전스: 가격 신고 but OBV 고점 낮아짐
    if (cur_close >= price_20_high * 0.98 and
            cur_obv < obv_20_high * 0.95):
        divergence = "bearish"
    # 강세 다이버전스: 가격 신저 but OBV 저점 높아짐
    elif (cur_close <= price_20_low * 1.02 and
              cur_obv > obv_20_low * 1.05):
        divergence = "bullish"

    # ④ OBV 브레이크아웃 (52봉 최고/최저 돌파)
    breakout = "none"
    if cur_obv > float(obv.iloc[-52:].max()) * 0.99:
        breakout = "bullish_breakout"
    elif cur_obv < float(obv.iloc[-52:].min()) * 1.01:
        breakout = "bearish_breakout"

    # 스코어 (-4 ~ +4)
    score = 0
    if obv_vs_ema == "bullish":          score += 1
    else:                                score -= 1
    if obv_trend in ("strong_up",):      score += 2
    elif obv_trend == "mild_up":         score += 1
    elif obv_trend == "mild_down":       score -= 1
    elif obv_trend == "strong_down":     score -= 2
    if divergence == "bullish":          score += 2
    elif divergence == "bearish":        score -= 2
    if breakout == "bullish_breakout":   score += 1
    elif breakout == "bearish_breakout": score -= 1

    return {
        "obv":         cur_obv,
        "obv_vs_ema":  obv_vs_ema,
        "obv_trend":   obv_trend,
        "divergence":  divergence,
        "breakout":    breakout,
        "score":       max(-4, min(4, score)),
    }
```

### OBV 핵심 신호 패턴

```
[신고가 돌파 + OBV 신고가 동반]  →  강한 상승 추세 확인  ★★★★★
[신고가 돌파 + OBV 미동반]       →  약세 다이버전스 경고  ⚠️
[OBV 52주 신고가 돌파]           →  기관 매집 신호        ★★★★
[OBV EMA 상향 돌파]              →  단기 매수 전환        ★★★
[OBV 지속 하락 중 가격 횡보]     →  분산(Distribution) 진행 중 ⚠️
```

## 지표 2: VWAP (Volume Weighted Average Price)

### 개념 및 계산

```python
def compute_vwap(df: pd.DataFrame,
                 reset: str = "daily") -> pd.DataFrame:
    """
    VWAP = Σ(Typical Price × Volume) / Σ(Volume)
    Typical Price = (High + Low + Close) / 3

    reset: "daily"   → 매일 리셋 (일중 기준)
           "weekly"  → 주간 기준
           "anchored"→ 특정 기점부터 누적 (AVWAP)
    """
    df = df.copy()
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['tp_vol']        = df['typical_price'] * df['volume']

    if reset == "daily":
        df['date'] = pd.to_datetime(df.index).date
        df['cum_tp_vol'] = df.groupby('date')['tp_vol'].cumsum()
        df['cum_vol']    = df.groupby('date')['volume'].cumsum()
    else:
        df['cum_tp_vol'] = df['tp_vol'].cumsum()
        df['cum_vol']    = df['volume'].cumsum()

    df['vwap'] = df['cum_tp_vol'] / df['cum_vol']

    # VWAP 표준편차 밴드 (±1σ, ±2σ)
    df['vwap_std'] = (
        df.groupby(df.get('date', df.index))['typical_price']
          .transform(lambda x: x.expanding().std())
          if reset == "daily"
          else df['typical_price'].expanding().std()
    )
    df['vwap_upper1'] = df['vwap'] + df['vwap_std']
    df['vwap_lower1'] = df['vwap'] - df['vwap_std']
    df['vwap_upper2'] = df['vwap'] + df['vwap_std'] * 2
    df['vwap_lower2'] = df['vwap'] - df['vwap_std'] * 2

    return df


def compute_anchored_vwap(df: pd.DataFrame,
                           anchor_idx: int) -> pd.Series:
    """
    AVWAP: 특정 이벤트(어닝, 급락, 신고가)부터 누적 VWAP
    기관의 평균 매입단가 추정에 활용
    """
    sub = df.iloc[anchor_idx:].copy()
    sub['tp']     = (sub['high'] + sub['low'] + sub['close']) / 3
    sub['tp_vol'] = sub['tp'] * sub['volume']
    avwap = sub['tp_vol'].cumsum() / sub['volume'].cumsum()
    return avwap
```

### VWAP 신호 분류

```python
def analyze_vwap(df: pd.DataFrame) -> dict:
    close  = float(df['close'].iloc[-1])
    vwap   = float(df['vwap'].iloc[-1])
    upper1 = float(df['vwap_upper1'].iloc[-1])
    lower1 = float(df['vwap_lower1'].iloc[-1])
    upper2 = float(df['vwap_upper2'].iloc[-1])
    lower2 = float(df['vwap_lower2'].iloc[-1])

    # 가격 위치 판단
    if close > upper2:
        position = "극단_과매수"
        signal   = "strong_sell"
    elif close > upper1:
        position = "과매수_구간"
        signal   = "sell"
    elif close > vwap:
        position = "VWAP_위_강세"
        signal   = "buy"
    elif close > lower1:
        position = "VWAP_아래_약세"
        signal   = "sell"
    elif close > lower2:
        position = "과매도_구간"
        signal   = "buy"
    else:
        position = "극단_과매도"
        signal   = "strong_buy"

    # VWAP 이격도
    disparity = round((close - vwap) / vwap * 100, 3)

    # VWAP 기울기 (추세 방향)
    vwap_series = df['vwap']
    slope_pct   = float(
        (vwap_series.iloc[-1] - vwap_series.iloc[-5]) /
        vwap_series.iloc[-5] * 100
    ) if len(vwap_series) > 5 else 0.0

    vwap_trend = (
        "uptrend"   if slope_pct > 0.1  else
        "downtrend" if slope_pct < -0.1 else
        "flat"
    )

    # 리전 크로스
    prev_close = float(df['close'].iloc[-2])
    prev_vwap  = float(df['vwap'].iloc[-2])
    cross = "none"
    if prev_close < prev_vwap and close > vwap:
        cross = "bullish_cross"
    elif prev_close > prev_vwap and close < vwap:
        cross = "bearish_cross"

    # 스코어 (-4 ~ +4)
    score_map = {
        "strong_buy": +3, "buy": +1,
        "sell": -1, "strong_sell": -3
    }
    score = score_map.get(signal, 0)
    if cross == "bullish_cross":   score += 1
    elif cross == "bearish_cross": score -= 1

    return {
        "vwap":       round(vwap, 4),
        "disparity":  disparity,
        "position":   position,
        "signal":     signal,
        "vwap_trend": vwap_trend,
        "cross":      cross,
        "score":      max(-4, min(4, score)),
        "bands": {
            "upper2": round(upper2, 4),
            "upper1": round(upper1, 4),
            "vwap":   round(vwap, 4),
            "lower1": round(lower1, 4),
            "lower2": round(lower2, 4),
        }
    }
```

### VWAP 핵심 신호 패턴

```
[가격 > VWAP + VWAP 우상향]         →  기관 매수 우위 (추세 추종 매수)    ★★★★
[VWAP 상향 돌파 + 거래량 급증]      →  강한 매수 전환 신호                ★★★★★
[가격 < VWAP + 반등 시도 실패]      →  저항 확인 → 추가 하락              ★★★
[+2σ 상단 도달]                      →  단기 과열 → 부분 익절              ★★★
[-2σ 하단 도달 + OBV 강세 다이버전스]→  강한 매수 기회                    ★★★★★
[AVWAP 하향 이탈]                    →  기관 손절 구간 진입 경고           ⚠️
```

## 지표 3: MFI (Money Flow Index)

### 개념 및 계산

```python
def compute_mfi(df: pd.DataFrame,
                period: int = 14) -> pd.DataFrame:
    """
    MFI = 거래량 가중 RSI
    = 100 - (100 / (1 + Money Flow Ratio))

    Typical Price 상승봉 → Positive Money Flow
    Typical Price 하락봉 → Negative Money Flow
    """
    df = df.copy()
    df['tp']  = (df['high'] + df['low'] + df['close']) / 3
    df['rmf'] = df['tp'] * df['volume']   # Raw Money Flow

    # 방향 구분
    tp_diff = df['tp'].diff()
    pos_mf  = df['rmf'].where(tp_diff > 0, 0)
    neg_mf  = df['rmf'].where(tp_diff < 0, 0)

    # 기간별 합산
    pos_sum = pos_mf.rolling(period).sum()
    neg_sum = neg_mf.rolling(period).sum().abs()

    mfr          = pos_sum / neg_sum.replace(0, np.nan)
    df['mfi']    = 100 - (100 / (1 + mfr))

    # MFI EMA (신호선)
    df['mfi_ema'] = df['mfi'].ewm(span=9, adjust=False).mean()
    return df
```

### MFI 신호 분류

```python
def analyze_mfi(df: pd.DataFrame,
                overbought: float = 80,
                oversold:   float = 20) -> dict:
    mfi       = df['mfi']
    mfi_ema   = df['mfi_ema']
    close     = df['close']

    cur_mfi   = float(mfi.iloc[-1])
    prev_mfi  = float(mfi.iloc[-2])
    cur_ema   = float(mfi_ema.iloc[-1])

    # ① 과매수/과매도
    if cur_mfi >= overbought:
        ob_os = "overbought"
    elif cur_mfi <= oversold:
        ob_os = "oversold"
    elif cur_mfi >= 60:
        ob_os = "bullish_zone"
    elif cur_mfi <= 40:
        ob_os = "bearish_zone"
    else:
        ob_os = "neutral"

    # ② MFI vs EMA 크로스
    cross = "none"
    if float(mfi.iloc[-2]) < float(mfi_ema.iloc[-2]) and cur_mfi > cur_ema:
        cross = "bullish_cross"
    elif float(mfi.iloc[-2]) > float(mfi_ema.iloc[-2]) and cur_mfi < cur_ema:
        cross = "bearish_cross"

    # ③ 가격-MFI 다이버전스
    lookback = 20
    p    = close.iloc[-lookback:]
    m    = mfi.iloc[-lookback:]
    p_c  = float(p.iloc[-1])
    m_c  = float(m.iloc[-1])

    divergence = "none"
    if p_c >= float(p.max()) * 0.98 and m_c < float(m.max()) * 0.90:
        divergence = "bearish"    # 가격 신고 + MFI 고점 낮아짐
    elif p_c <= float(p.min()) * 1.02 and m_c > float(m.min()) * 1.10:
        divergence = "bullish"    # 가격 신저 + MFI 저점 높아짐

    # ④ 실패 스윙 (Failure Swing)
    failure = "none"
    if len(mfi) >= 6:
        r = mfi.iloc[-6:]
        # 매수 실패 스윙: OB → 하락 → 상승 but OB 미도달 → 하락
        if (r.iloc[0] > overbought and r.iloc[-1] < r.iloc[-3]
                and r.iloc[-3] < overbought):
            failure = "sell_failure_swing"
        # 매도 실패 스윙: OS → 상승 → 하락 but OS 미도달 → 상승
        elif (r.iloc[0] < oversold and r.iloc[-1] > r.iloc[-3]
                  and r.iloc[-3] > oversold):
            failure = "buy_failure_swing"

    # 스코어 (-4 ~ +4)
    score = 0
    ob_score = {
        "oversold": +2, "bullish_zone": +1,
        "neutral": 0,
        "bearish_zone": -1, "overbought": -2
    }
    score += ob_score.get(ob_os, 0)
    if cross == "bullish_cross":          score += 1
    elif cross == "bearish_cross":        score -= 1
    if divergence == "bullish":           score += 2
    elif divergence == "bearish":         score -= 2
    if failure == "buy_failure_swing":    score += 1
    elif failure == "sell_failure_swing": score -= 1

    return {
        "mfi":        round(cur_mfi, 2),
        "ob_os":      ob_os,
        "cross":      cross,
        "divergence": divergence,
        "failure":    failure,
        "score":      max(-4, min(4, score)),
    }
```

### MFI vs RSI 차이점

```
구분          RSI                    MFI
────────────────────────────────────────────────────
계산 기반    순수 가격 변화         가격 × 거래량 (자금 흐름)
민감도       가격 급변에 민감        거래량 급변에 민감
신호 특성    추세 강도 측정          매집/분산 탐지

과매도       30 (표준)              20 (더 엄격)
핵심 활용    다이버전스, 50선 크로스  거래량 동반 여부 확인
```

## 거래량 복합 분석: OBV + VWAP + MFI 통합

```python
import numpy as np
import pandas as pd
from dataclasses import dataclass


@dataclass
class VolumeSignal:
    obv_score:    int
    vwap_score:   int
    mfi_score:    int
    composite:    float
    regime:       str
    action:       str
    confidence:   float
    details:      dict


def volume_composite_signal(df: pd.DataFrame) -> VolumeSignal:
    """
    OBV + VWAP + MFI 복합 거래량 신호 산출
    """
    # 각 지표 계산
    df = compute_obv(df)
    df = compute_vwap(df)
    df = compute_mfi(df)

    obv_r  = analyze_obv(df)
    vwap_r = analyze_vwap(df)
    mfi_r  = analyze_mfi(df)

    # 가중 합산 (VWAP 40% / OBV 35% / MFI 25%)
    w_obv, w_vwap, w_mfi = 0.35, 0.40, 0.25

    obv_norm  = obv_r['score']  / 4.0
    vwap_norm = vwap_r['score'] / 4.0
    mfi_norm  = mfi_r['score']  / 4.0

    raw_score = (obv_norm  * w_obv  +
                 vwap_norm * w_vwap +
                 mfi_norm  * w_mfi)

    composite = round(raw_score * 10, 3)   # [-10, +10]

    # 레짐 분류
    regime = _classify_volume_regime(obv_r, vwap_r, mfi_r)

    # 액션
    action = (
        "STRONG_BUY"  if composite >= 4  else
        "BUY"         if composite >= 2  else
        "WEAK_BUY"    if composite >= 1  else
        "HOLD"        if composite >= -1 else
        "WEAK_SELL"   if composite >= -2 else
        "SELL"        if composite >= -4 else
        "STRONG_SELL"
    )

    # 신뢰도: 3개 지표 방향 일치도
    scores    = [obv_r['score'], vwap_r['score'], mfi_r['score']]
    pos_count = sum(1 for s in scores if s > 0)
    neg_count = sum(1 for s in scores if s < 0)
    confidence = abs(pos_count - neg_count) / 3.0

    return VolumeSignal(
        obv_score=obv_r['score'],
        vwap_score=vwap_r['score'],
        mfi_score=mfi_r['score'],
        composite=composite,
        regime=regime,
        action=action,
        confidence=round(confidence, 3),
        details={
            "obv":  obv_r,
            "vwap": vwap_r,
            "mfi":  mfi_r,
        }
    )


def _classify_volume_regime(obv_r, vwap_r, mfi_r) -> str:
    """
    거래량 3지표 기반 시장 국면 분류
    """
    # 매집 (Accumulation)
    if (obv_r['divergence'] == "bullish" or
            mfi_r['divergence'] == "bullish"):
        if vwap_r['position'] in ("VWAP_아래_약세", "과매도_구간"):
            return "accumulation"

    # 분산 (Distribution)
    if (obv_r['divergence'] == "bearish" or
            mfi_r['divergence'] == "bearish"):
        if vwap_r['position'] in ("VWAP_위_강세", "과매수_구간"):
            return "distribution"

    # 브레이크아웃
    if (obv_r['breakout'] == "bullish_breakout" and
            vwap_r['cross'] == "bullish_cross"):
        return "bullish_breakout"

    if (obv_r['breakout'] == "bearish_breakout" and
            vwap_r['cross'] == "bearish_cross"):
        return "bearish_breakout"

    # 강세 추세
    if (obv_r['obv_trend'] in ("strong_up", "mild_up") and
            vwap_r['position'] == "VWAP_위_강세" and
            mfi_r['ob_os'] in ("bullish_zone", "overbought")):
        return "strong_bull"

    # 약세 추세
    if (obv_r['obv_trend'] in ("strong_down", "mild_down") and
            vwap_r['position'] == "VWAP_아래_약세" and
            mfi_r['ob_os'] in ("bearish_zone", "oversold")):
        return "strong_bear"

    return "neutral"
```

## 레짐별 거래량 전략 매핑

```
거래량 레짐              핵심 신호 조합                        전략
────────────────────────────────────────────────────────────────────
accumulation       OBV 강세 다이버전스 + MFI 강세 다이버전스   분할 매수 준비
distribution       OBV 약세 다이버전스 + MFI 약세 다이버전스   부분 익절 / 헤지
bullish_breakout   OBV 신고가 + VWAP 상향 크로스 + MFI > 60   추세 추종 매수
bearish_breakout   OBV 신저가 + VWAP 하향 크로스 + MFI < 40   추세 추종 매도
```

## 전체 시스템 연동: 가격 + 거래량 통합 스코어

```python
def full_composite_signal(price_signal,       # CompositeSignal (ATR/BB/MACD/RSI/MA)
                           vol_signal:  VolumeSignal,
                           fib_signal          # FibSignal
                           ) -> dict:
    """
    가격 기반 지표 (60%) + 거래량 지표 (30%) + 피보나치 (10%) 통합
    """
    price_score = price_signal.composite_score   # [-10, +10]
    vol_score   = vol_signal.composite             # [-10, +10]
    fib_boost   = 1.2 if fib_signal.confidence == "HIGH" else 1.0

    final_score = round(
        (price_score * 0.60 +
         vol_score   * 0.30) * fib_boost,
        3
    )
    final_score = max(-10, min(10, final_score))

    # 신호 일치 확인 (가격 + 거래량 방향 일치 여부)
    price_dir = 1 if price_score > 0 else -1 if price_score < 0 else 0
    vol_dir   = 1 if vol_score   > 0 else -1 if vol_score   < 0 else 0
    confirmed = (price_dir == vol_dir and price_dir != 0)

    action = (
        "STRONG_BUY"  if final_score >= 5  and confirmed else
        "BUY"         if final_score >= 2  and confirmed else
        "WEAK_BUY"    if final_score >= 1  else
        "HOLD"        if final_score >= -1 else
        "WEAK_SELL"   if final_score >= -2 else
        "SELL"        if final_score >= -5 and confirmed else
        "STRONG_SELL" if confirmed else
        "UNCONFIRMED"  # 가격/거래량 방향 불일치
    )

    return {
        "price_score":  price_score,
        "vol_score":    vol_score,
        "fib_boost":    fib_boost,
        "final_score":  final_score,
        "confirmed":    confirmed,
        "action":       action,
        "vol_regime":   vol_signal.regime,
        "price_regime": price_signal.regime.value,
    }
```

## 실전 주의사항

1. **VWAP 타임프레임 구분**: 일중 단타 → 일간 리셋 VWAP / 스윙 → 주간/AVWAP 사용
2. **OBV 절대값 무의미**: OBV는 방향과 기울기만 의미 있음 — 종목 간 절대값 비교 불가
3. **MFI 80/20 임계값**: RSI보다 엄격한 기준 적용 — 70/30 사용 시 오신호 증가
4. **거래량 이상 급증 필터**: 배당락/결산일/지수 편출입 당일 거래량은 신호에서 제외
5. **3지표 방향 불일치 시 관망**: OBV ↑ + VWAP ↓ + MFI 중립 → 방향성 미확인 → 진입 금지
